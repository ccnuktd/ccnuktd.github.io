3.定义指向函数的指针

```c++
//一个指向函数的指针，返回值为void，参数void
//以下是鼠标编辑器的操作
typedef void (* PF) ();
PF edit_ops[] = {&cut,&paste,&copy,&search};
PF file_ops[] = {
	&open,&append,&close,&write
};
PF* button1 = &editops[0];
PF* button2 = file_ops;
//调用按键2的第3个参数
button2[2]();
```
